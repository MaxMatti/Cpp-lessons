\begin{frame}[fragile]
    \frametitle{Reading Input}
\begin{lstlisting}
#include <iostream>
#include <string>
using namespace std;

int main() {
    cout << "Please enter your name: ";
    string name;
    cin >> name;
    cout << "Hello " << name << "!\n";
    return 0;
}
\end{lstlisting}
\end{frame}
